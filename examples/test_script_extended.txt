// Variable assignment and basic arithmetic
x = 5
y = x * 2 + 3
print(x)  // Should print 5
print(y)  // Should print 13

// Conditional (if-else)
if (x > 3) {
    print(x)  // Should print 5
} else {
    print(0)
}

// While loop with print
while (x < 10) {
    x = x + 1
    print(x)  // Should print 6, 7, 8, 9, 10
}

// Function definition and call
def add(a, b) {
    return a + b
}
z = add(10, 20)
print(z)  // Should print 30

// Nested functions and conditionals
if (add(1, 2) == 3) {
    print(100)  // Should print 100
} else {
    print(200)
}

// Trigonometric functions (sin, cos, etc.)
a = sin(0)
b = cos(0)
print(a)  // Should print 0.0
print(b)  // Should print 1.0

// More complex expressions with sin, cos, and power
angle = 45
radians = angle * 3.14159 / 180  // Convert degrees to radians
c = sin(radians)
print(c)  // Should print the sine of 45 degrees (0.707... approx)

// Use power operator (^)
power_result = 2 ^ 3  // 2 to the power of 3
print(power_result)  // Should print 8 (2^3 = 8)

// Using functions with other functions
def multiply(a, b) {
    return a * b
}
result = multiply(add(2, 3), 4)
print(result)  // Should print 20, since (2 + 3) * 4 = 20
